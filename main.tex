\begin{frame}{System Model}

\begin{itemize}

\item Header success probability (single channel):
\[
p_{ok,H} = e^{-\left[2\lambda_H T_H + \lambda_P (T_H + T_P)\right]}
\]

\item Payload success probability (single channel):
\[
p_{ok,P} = e^{-\left[2\lambda_P T_P + \lambda_H (T_H + T_P)\right]}
\]

\item Header success probability with LR-FHSS hopping:
\[
p_{ok,H} = e^{-\frac{1}{C}\left[2\lambda_H T_H + \lambda_P (T_H + T_P)\right]}
\]

\item Payload success probability with LR-FHSS hopping:
\[
p_{ok,P} = e^{-\frac{1}{C}\left[2\lambda_P T_P + \lambda_H (T_H + T_P)\right]}
\]

\end{itemize}


\end{frame}

\begin{frame}{System Model}
\begin{itemize}

\item Probability of receiving at least one Header replica:
\[
P_H = 1 - (1 - p_{ok,H})^{N_H}
\]

\item Probability of receiving \(k\) Payload fragments:
\[
p_{ok,P}(X = k) =
\binom{N_P}{k}
p_{ok,P}^{\,k}
(1 - p_{ok,P})^{N_P-k}
\]

\item Probability of receiving at least \(CR \times N_P\) Payload fragments:
\[
P_P =
\sum_{k = CR \times N_P}^{N_P}
p_{ok,P}(X = k)
\]

\item Overall packet success probability:
\[
P_{\text{success}} = P_H \times P_P
\]

\item Expected number of successfully transmitted packets:
\[
N_{bTr} = \lambda N \, P_{\text{success}}
\]

\end{itemize}
    
\end{frame}

\begin{frame}{Comparative Study}

\begin{itemize}

\item State-of-the-art models use the \textbf{balls-into-bins} (Classic Occupancy Problem).

\item Success probability of a fragment:
\[
p_{ok} =
\left(1 - \frac{1}{C}\right)^{A-1}
\]

\item Since the number of arrivals \(A\) is unknown, it is approximated by its expectation:

\[
p_{ok} =
\left(\frac{C-1}{C}\right)^{E(A)-1}
=
\left(\frac{C-1}{C}\right)^{2\lambda T - 1}
\]

\item \textbf{Limitation:}
For low traffic (\(2\lambda T < 1\)), the model can give
\[
p_{ok} > 1
\]
which is not valid for a probability.

\end{itemize}
    
\end{frame}

\section{Parameters Optimization}

\begin{frame}{Optimal Header Replicas}

\begin{itemize}
\item The proposed model allows finding the optimal value of a parameter analytically by differentiating the expected number of successful transmissions:
\[
N_{bTr} = f(N_H, N_P, N, \dots)
\]

\item Example: Optimal number of header replicas \(N_H\)
  \begin{itemize}
    \item Derivative \(\frac{d N_{bTr}}{d N_H}\) is complex (exponentials from Poisson probabilities)
    \item Solved numerically → derivative vanishes at \(N_H = 3\)
  \end{itemize}


\item Approach can be extended to other parameters: \(N_P\), number of devices \(N\), etc., via derivative + root-finding.
\end{itemize}

\end{frame}

\section{Limitations}

\begin{frame}{Model Assumptions and Experiments}

\begin{itemize}
\item Analytical model assumes fragments belong to \textbf{independent Poisson flows}:
  \begin{itemize}
    \item Arrival rate = packet arrival rate × number of each fragment type.
  \end{itemize}

\item In reality, fragments are generated in \textbf{batches}:
  \begin{itemize}
    \item One packet → \(N_H\) headers + \(N_P\) payload fragments.
    \item Fragments from the same packet cannot collide (device transmits one fragment at a time).
  \end{itemize}

\item Experiments conducted:
  \begin{enumerate}
    \item Fixed nodes \(N = 125\), vary packet generation rate \(\lambda\)
    \item Fixed \(\lambda = 0.05\), vary number of nodes
  \end{enumerate}
\end{itemize}

\end{frame}


\begin{frame}{Observations and Limitations}

\begin{itemize}
\item Observations:
  \begin{itemize}
    \item Node-varying experiment → lower success rate (aligns with model assumptions)
    \item \(\lambda\)-varying experiment → higher success rate (fragments from same node, limited by device transmission rate)
  \end{itemize}

\item \textbf{Main limitation:} 
  \begin{itemize}
    \item Model ignores batch arrivals and “no self-collision” property.
    \item Under-predicts performance under heavy load.
    \item Over-predicts under light load.
  \end{itemize}
\end{itemize}

\end{frame}


\section{Conclusion}

\begin{frame}{Conclusion}

\begin{itemize}
\item Presented a new analytical framework: \textbf{MC²ABM} (Multi-Class Multi-Channel ALOHA Based Model) for LR-FHSS networks.
\item Addresses limitations of previous models:
  \begin{itemize}
    \item Overestimation of collision probabilities in worst-case scenarios.
  \end{itemize}
\item Key advantages:
  \begin{itemize}
    \item High accuracy across various network densities.
    \item Outperforms existing models in precision.
    \item Enables parameter optimization (e.g., number of header replicas) without costly experiments.
  \end{itemize}
\item \textbf{Future work:} Improve model accuracy and robustness; develop strategies for achieving target QoS in satellite communication scenarios.
\end{itemize}

\end{frame}
